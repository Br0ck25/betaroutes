// 📦 Cloudflare Worker: user-auth + per-user log storage in KV export default { async fetch(request, env, ctx) { const url = new URL(request.url); const method = request.method; // Route: /api/signup if (url.pathname === "/api/signup" && method === "POST") { const { username, password } = await request.json(); if (!username || !password) return new Response("Missing username or password", { status: 400 }); const existing = await env.LOGS_KV.get(`user:${username}`); if (existing) return new Response("Username already exists", { status: 409 }); const hashed = await hashPassword(password); await env.USER_STORE.put(`user:${username}`, hashed); return json({ token: createToken(username) }); } // Route: /api/login if (url.pathname === "/api/login" && method === "POST") { const { username, password } = await request.json(); const storedHash = await env.USER_STORE.get(`user:${username}`); if (!storedHash) return new Response("Invalid username", { status: 404 }); const valid = await verifyPassword(password, storedHash); if (!valid) return new Response("Invalid password", { status: 401 }); return json({ token: createToken(username) }); } // Authenticated routes: const auth = request.headers.get("Authorization"); const user = verifyToken(auth); if (!user) return new Response("Unauthorized", { status: 401 }); // GET logs if (url.pathname === "/logs" && method === "GET") { const data = await env.USER_STORE.get(`logs:${user}`); return json(JSON.parse(data || "[]")); } // POST logs if (url.pathname === "/logs" && method === "POST") { const logs = await request.json(); await env.USER_STORE.put(`logs:${user}`, JSON.stringify(logs)); return json({ status: "ok" }); } return new Response("Not found", { status: 404 }); }, }; // Helpers function json(obj) { return new Response(JSON.stringify(obj), { headers: { "Content-Type": "application/json" }, }); } // 🔐 Token functions (simplified - not secure for production) function createToken(username) { return btoa(username + ":token"); } function verifyToken(token) { try { const decoded = atob(token || ""); const [username] = decoded.split(":"); return username; } catch { return null; } } // 🔐 Password hashing using SubtleCrypto async function hashPassword(password) { const data = new TextEncoder().encode(password); const hash = await crypto.subtle.digest("SHA-256", data); return btoa(String.fromCharCode(...new Uint8Array(hash))); } async function verifyPassword(password, hash) { const hashed = await hashPassword(password); return hashed === hash; }
